CREATE TABLE Categories (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    Name VARCHAR(100) NOT NULL
);

CREATE TABLE Vehicles (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    ImageUrl VARCHAR(255),
    Brand VARCHAR(100),
    Model VARCHAR(100),
    Mileage INT,
    Years INT,
    Price INT,
    CategoryId INT,
    FOREIGN KEY (CategoryId) REFERENCES Categories(Id)
);
